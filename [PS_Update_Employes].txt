CREATE PROCEDURE PS_Update_Employes   
@Matricule	varchar(50),
@Nom	varchar(50)	,
@Prenom	varchar(50)	,
@localisation	varchar(50)	,
@Date_Naiss	datetime	,
@Date_Entree	datetime	,
@Salaire_Mensuel	decimal(18, 0)	,
@Adresse	varchar(50)	,
@IdDepartement	int	,
@Poste	varchar(50)	,
@photo	varchar(50)	,
@IdEntreprise	int	,
@sex	varchar(50)	,
@nationalite	varchar(15)	,
@Sit_Familial	varchar(50)	,
@BP	decimal(18, 0)	,
@PermisConduire	char(10)	,
@commentaire	varchar(100),
@Email			varchar(50),
@oldMatricule varchar(50),
@Diplome varchar(50),
@DEF varchar(50)
 AS
BEGIN
	        UPDATE Employes SET Matricule=@Matricule,
								      Nom=@Nom,
								   Prenom=@Prenom,
						     localisation=@localisation,
							   Date_Naiss=@Date_Naiss,
							  Date_Entree=@Date_Entree,
						  Salaire_Mensuel=@Salaire_Mensuel,
								  Adresse=@Adresse,
						    IdDepartement=@IdDepartement,
								    Poste=@Poste,
								    photo=@photo,
							 IdEntreprise=@IdEntreprise,
							          sex=@sex,
							  nationalite=@nationalite,
							 Sit_Familial=@Sit_Familial,
								       BP=@BP,
						   PermisConduire=@PermisConduire,
							  commentaire=@commentaire,
								    Email=@Email,
								Diplome=@Diplome,
									DEF=@DEF
          WHERE Matricule=@oldMatricule					
											
END
