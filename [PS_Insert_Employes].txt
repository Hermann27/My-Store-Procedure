CREATE PROCEDURE PS_Insert_Employes 
@Matricule	varchar(50),
@Nom	varchar(50)	,
@Prenom	varchar(50)	,
@localisation	varchar(50)	,
@Date_Naiss	datetime	,
@Date_Entree	datetime	,
@Salaire_Mensuel	decimal(18, 0)	,
@Adresse	varchar(50)	,
@IdDepartement	int	,
@Poste	varchar(50)	,
@photo	varchar(50)	,
@IdEntreprise	int	,
@sex	varchar(50)	,
@nationalite	varchar(15)	,
@Sit_Familial	varchar(50)	,
@BP	decimal(18, 0)	,
@PermisConduire	char(10)	,
@commentaire	varchar(100),
@Email varchar(50),
@Diplome varchar(50) ,
@DEF varchar(50)
 AS
BEGIN
	INSERT INTO Employes(Matricule,
								Nom,
								Prenom,
								localisation,
								Date_Naiss,
								Date_Entree,
								Salaire_Mensuel,
								Adresse,
								IdDepartement,
								Poste,
								photo,
								IdEntreprise,
								sex,
								nationalite,
								Sit_Familial,
								BP,
								PermisConduire,
								commentaire,
								Email,
								Diplome,
								DEF
								)values(@Matricule,
											@Nom,
											@Prenom,
											@localisation,
											@Date_Naiss,
											@Date_Entree,
											@Salaire_Mensuel,
											@Adresse,
											@IdDepartement,
											@Poste,
											@photo,
											@IdEntreprise,
											@sex,
											@nationalite,
											@Sit_Familial,
											@BP,
											@PermisConduire,
											@commentaire,
											@Email,
											@Diplome,
											@DEF
											)
END
